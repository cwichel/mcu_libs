129 int16_t USB_recv_control(void* d, int16_t len) {
130  int16_t length = len;
131  while(length) {
132  // Dont receive more than the USB Control EP has to offer
133  // Use fixed 64 because control EP always have 64 bytes even on 16u2.
134  int16_t recvLength = length;
135  if(recvLength > 64) {
136  recvLength = 64;
137  }
138 
139  // Write data to fit to the end (not the beginning) of the array
140  wait_out();
141  recv((uint8_t*)d + len - length, recvLength);
142  clear_out();
143  length -= recvLength;
144  }
145  return len;

163 // Blocking Send of data to an endpoint
164 int16_t USB_send(uint8_t ep, const void* d, int16_t len) {
165  if (!_usbConfiguration)
166  return -1;
167 
168  if (_usbSuspendState & (1<<SUSPI)) {
169  //send a remote wakeup
170  UDCON |= (1 << RMWKUP);
171  }
172 
173  int16_t r = len;
174  const uint8_t* data = (const uint8_t*)d;
175  uint8_t timeout = 250; // 250ms timeout on send? TODO
176  bool sendZlp = false;
177 
178  while (len || sendZlp) {
179  uint8_t n = USB_send_space(ep);
180  if (n == 0) {
181  if (!(--timeout))
182  return -1;
183  _delay_ms(1);
184  continue;
185  }
186 
187  if (n > len) {
188  n = len;
189  }
190 
191  {
192  LockEP lock(ep);
193  // Frame may have been released by the SOF interrupt handler
194  if (!rw_allowed())
195  continue;
196  len -= n;
197 
198  if (ep & TRANSFER_ZERO) {
199  while (n--)
200  send8(0);
201  } else if (ep & TRANSFER_PGM) {
202  while (n--)
203  send8(pgm_read_byte(data++));
204  } else {
205  while (n--)
206  send8(*data++);
207  }
208  if (sendZlp) {
209  release_TX();
210  sendZlp = false;
211  } else if (!rw_allowed()) { // ...release if buffer is full...
212  release_TX();
213  if (len == 0) sendZlp = true;
214  } else if ((len == 0) && (ep & TRANSFER_RELEASE)) { // ...or if forced with TRANSFER_RELEASE
215  // XXX: TRANSFER_RELEASE is never used can be removed?
216  release_TX();
217  }
218  }
219  }
220  return r;
221 }

226  if (!_usbConfiguration || len < 0)
227  return -1;

229  uint8_t n = fifo_byte_count();
230  len = min(n, len);
231  n = len;
232  uint8_t* dst = (uint8_t*)d;

280 bool USBDevice::wakeup_host() {
281  // clear any previous wakeup request which might have been set but could be processed at that time
282  // e.g. because the host was not suspended at that time
283  UDCON &= ~(1 << RMWKUP);
284 
285  if(!(UDCON & (1 << RMWKUP))
286  && (_usbSuspendState & (1<<SUSPI))
287  && (_usbCurrentStatus & FEATURE_REMOTE_WAKEUP_ENABLED)) {
288  // This short version will only work, when the device has not been suspended. Currently the
289  // Arduino core doesn't handle SUSPEND at all, so this is ok.
290  USB_clock_enable();

300 // General interrupt
301 ISR(USB_GEN_vect) {
302  uint8_t udint = UDINT;
303  UDINT &= ~((1<<EORSTI) | (1<<SOFI)); // clear the IRQ flags for the IRQs which are handled here, except WAKEUPI and SUSPI (see below)
304 
305  // End of Reset
306  if (udint & (1<<EORSTI)) {
307  EP_init(0,EP_TYPE_CONTROL,EP_SINGLE_64); // init ep0
308  _usbConfiguration = 0; // not configured yet
309  UEIENX = 1 << RXSTPE; // Enable interrupts for ep0
310  }
311  // Start of Frame - happens every millisecond so we use it for TX and RX LED one-shot timing, too
312  if (udint & (1<<SOFI)) {
313  USB_flush(CDC_TX); // Send a tx frame if found
314  }
315  // the WAKEUPI interrupt is triggered as soon as there are non-idle patterns on the data
316  // lines. Thus, the WAKEUPI interrupt can occur even if the controller is not in the "suspend" mode.
317  // Therefore the we enable it only when USB is suspended
318  if (udint & (1<<WAKEUPI)) {
319  UDIEN = (UDIEN & ~(1<<WAKEUPE)) | (1<<SUSPE); // Disable interrupts for WAKEUP and enable interrupts for SUSPEND
320  //TODO
321  // WAKEUPI shall be cleared by software (USB clock inputs must be enabled before).
322  //USB_clock_enable();
323  UDINT &= ~(1<<WAKEUPI);
324  _usbSuspendState = (_usbSuspendState & ~(1<<SUSPI)) | (1<<WAKEUPI);
325  } else if (udint & (1<<SUSPI)) { // only one of the WAKEUPI / SUSPI bits can be active at time
326  UDIEN = (UDIEN & ~(1<<SUSPE)) | (1<<WAKEUPE); // Disable interrupts for SUSPEND and enable interrupts for WAKEUP
327  //TODO
328  //USB_clock_disable();
329  UDINT &= ~((1<<WAKEUPI) | (1<<SUSPI)); // clear any already pending WAKEUP IRQs and the SUSPI request
330  _usbSuspendState = (_usbSuspendState & ~(1<<WAKEUPI)) | (1<<SUSPI);
331  }

343 // Endpoint 0 interrupt
344 ISR(USB_COM_vect) {
345  EP_set(0);
346  if (!received_setup_int())
347  return;
348 
349  USBSetup setup;
350  recv((uint8_t*)&setup,8);
351  clear_setup_int();
352 
353  uint8_t requestType = setup.bmRequestType;
354  if (requestType & REQUEST_DEVICETOHOST)
355  wait_in();
356  else
357  clear_in();
358 
359  bool ok = true;
360  if (REQUEST_STANDARD == (requestType & REQUEST_TYPE)) {
361  // Standard Requests
362  uint8_t r = setup.bRequest;
363  uint16_t wValue = setup.wValueL | (setup.wValueH << 8);
364  if (GET_STATUS == r) {
365  if (requestType == (REQUEST_DEVICETOHOST | REQUEST_STANDARD | REQUEST_DEVICE)) {
366  send8(_usbCurrentStatus);
367  send8(0);
368  } else {
369  // TODO: handle the HALT state of an endpoint here
370  // see "Figure 9-6. Information Returned by a GetStatus() Request to an Endpoint" in usb_20.pdf for more information
371  send8(0);
372  send8(0);
373  }
374  } else if (CLEAR_FEATURE == r) {
375  if((requestType == (REQUEST_HOSTTODEVICE | REQUEST_STANDARD | REQUEST_DEVICE))
376  && (wValue == DEVICE_REMOTE_WAKEUP)) {
377  _usbCurrentStatus &= ~FEATURE_REMOTE_WAKEUP_ENABLED;
378  }
379  } else if (SET_FEATURE == r) {
380  if((requestType == (REQUEST_HOSTTODEVICE | REQUEST_STANDARD | REQUEST_DEVICE))
381  && (wValue == DEVICE_REMOTE_WAKEUP)) {
382  _usbCurrentStatus |= FEATURE_REMOTE_WAKEUP_ENABLED;
383  }
384  } else if (SET_ADDRESS == r) {
385  wait_in();
386  UDADDR = setup.wValueL | (1<<ADDEN);
387  } else if (GET_DESCRIPTOR == r) {
388  ok = send_descriptor(setup);
389  } else if (SET_DESCRIPTOR == r) {
390  ok = false;
391  } else if (GET_CONFIGURATION == r) {
392  send8(1);
393  } else if (SET_CONFIGURATION == r) {
394  if (REQUEST_DEVICE == (requestType & REQUEST_RECIPIENT)) {
395  EP_init();
396  _usbConfiguration = setup.wValueL;
397  } else {
398  ok = false;
399  }
400  } else if (GET_INTERFACE == r) {
401  } else if (SET_INTERFACE == r) {
402  }
403  } else {
404  init_control(setup.wLength); // Max length of transfer
405  ok = class_interface_request(setup);
406  }
407  if (ok) {
408  clear_in();
409  } else {
410  stall();
411  }
412 }

537 static bool send_descriptor(USBSetup& setup) {
538  int16_t ret;
539  uint8_t t = setup.wValueH;
540  if (USB_CONFIGURATION_DESCRIPTOR_TYPE == t)
541  return send_configuration(setup.wLength);
542 
543  init_control(setup.wLength);
544  #ifdef PLUGGABLE_USB_ENABLED
545  ret = PluggableUSB().get_descriptor(setup);
546  if (ret != 0) {
547  return (ret > 0 ? true : false);
548  }
549  #endif
550  const uint8_t* desc_addr = 0;
551  if (USB_DEVICE_DESCRIPTOR_TYPE == t) {
552  desc_addr = (const uint8_t*)&USB_DeviceDescriptorIAD;
553  } else if (USB_STRING_DESCRIPTOR_TYPE == t) {
554  if (setup.wValueL == 0) {
555  desc_addr = (const uint8_t*)&STRING_LANGUAGE;
556  } else if (setup.wValueL == IPRODUCT) {
557  return USB_send_string_descriptor(STRING_PRODUCT, strlen(USB_PRODUCT), TRANSFER_PGM);
558  } else if (setup.wValueL == IMANUFACTURER) {
559  return USB_send_string_descriptor(STRING_MANUFACTURER, strlen(USB_MANUFACTURER), TRANSFER_PGM);
560  } else if (setup.wValueL == ISERIAL) {
561  #ifdef PLUGGABLE_USB_ENABLED
562  char name[ISERIAL_MAX_LEN];
563  PluggableUSB().get_short_name(name);
564  return USB_send_string_descriptor((uint8_t*)name, strlen(name), 0);
565  #endif
566  } else {
567  return false;
568  }
569  }
570  if (desc_addr == 0)
571  return false;
572  uint8_t desc_length = pgm_read_byte(desc_addr);
573  USB_send_control(TRANSFER_PGM,desc_addr,desc_length);
574  return true;
575 }

607 static inline void USB_clock_enable(void) {
608  #if defined(UHWCON)
609  UHWCON |= (1<<UVREGE); // power internal reg
610  #endif
611  USBCON = (1<<USBE) | (1<<FRZCLK); // clock frozen, usb enabled
612  #if defined(PINDIV)
613  // ATmega32U4
614  #if F_CPU == 16000000UL
615  PLLCSR |= (1<<PINDIV); // Need 16 MHz xtal
616  #elif F_CPU == 8000000UL
617  PLLCSR &= ~(1<<PINDIV); // Need 8 MHz xtal
618  #else
619  #error "Clock rate of F_CPU not supported"
620  #endif
621  #elif defined(__AVR_AT90USB82__) || defined(__AVR_AT90USB162__) || defined(__AVR_ATmega32U2__) || defined(__AVR_ATmega16U2__) || defined(__AVR_ATmega8U2__)
622  // for the u2 Series the datasheet is confusing. On page 40 its called PINDIV and on page 290 its called PLLP0
623  #if F_CPU == 16000000UL
624  // Need 16 MHz xtal
625  PLLCSR |= (1 << PLLP0);
626  #elif F_CPU == 8000000UL
627  // Need 8 MHz xtal
628  PLLCSR &= ~(1 << PLLP0);
629  #endif
630  #elif defined(PLLP2)
631  // AT90USB646, AT90USB647, AT90USB1286, AT90USB1287
632  #if F_CPU == 16000000UL
633  #if defined(__AVR_AT90USB1286__) || defined(__AVR_AT90USB1287__)
634  // For Atmel AT90USB128x only. Do not use with Atmel AT90USB64x.
635  PLLCSR = (PLLCSR & ~(1<<PLLP1)) | ((1<<PLLP2) | (1<<PLLP0)); // Need 16 MHz xtal
636  #elif defined(__AVR_AT90USB646__) || defined(__AVR_AT90USB647__)
637  // For AT90USB64x only. Do not use with AT90USB128x.
638  PLLCSR = (PLLCSR & ~(1<<PLLP0)) | ((1<<PLLP2) | (1<<PLLP1)); // Need 16 MHz xtal
639  #else

686 // Send a USB descriptor string. The string is stored in PROGMEM as a
687 // plain ASCII string but is sent out as UTF-16 with the correct 2-byte
688 // prefix
689 static bool USB_send_string_descriptor(const uint8_t*string_P, uint8_t string_len, uint8_t flags) {
690  send_control(2 + string_len * 2);
691  send_control(3);
692  bool pgm = flags & TRANSFER_PGM;
693  for(uint8_t i = 0; i < string_len; i++) {
694  bool r = send_control(pgm ? pgm_read_byte(&string_P[i]) : string_P[i]);
695  r &= send_control(0); // high byte
696  if(!r) {
697  return false;
698  }
699  }
700  return true;
701 }
