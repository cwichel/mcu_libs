186 int16_t Stream::peek_next_digit(stream_look_mode_t look_mode, bool is_decimal) {
187  int16_t byte;
188 
189  while (true) {
190  // Get the next byte (if any)
191  byte = timed_peek();
192 
193  // If there's no byte available or the byte is numeric return the byte
194  if (
195  (byte < 0) || // No byte or timeout reached (error -1)
196  (byte == '-') || // Negative value
197  ((byte >= '0') && (byte <= '9')) || // Numeric char
198  (is_decimal && (byte == '.')) // Decimal point
199  )
200  return byte;
201 
202  // If the method reach this point that means that the byte is non numeric.
203  // This checks if the detected byte has to be skipped.
204  switch (look_mode) {
205  case SKIP_NONE: return -1; break; // Returns that no numeric byte was detected.
206  case SKIP_ALL: break; // Just skip the byte (until a numeric byte is detected).
207  case SKIP_WHITESPACE: {
208  switch (byte) {
209  case ' ':
210  case '\t':
211  case '\r':
212  case '\n': break; // If any of this bytes is detected, skip.
213  default: return -1; break; // Return that no numeric byte was detected.
214  }
215  } break;
216  }
217 
218  // Reading the buffer will "skip" the byte.
219  read();
220  }
221 }
