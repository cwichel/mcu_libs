167 void UART::flush(void) {
168  // If the UART was never used to send a byte, return.
169  if (!_used)
170  return;
171  // Otherwise, wait until the writing is completed.
172  // While a transmission is completed or the tx data interrupt is active
173  while (_ctrl->stat.field.txc || _ctrl->ctrl1.field.udrie) {
174  // Check if the interruptions are on
175  if (bit_is_clear(SREG, SREG_I) && _ctrl->ctrl1.field.udrie)
176  // If the buffer is available, write
177  if (_ctrl->stat.field.udre) _tx_callback();
178  }
179 }

181 size_t UART::write(uint8_t byte) {
182  uint8_t idx;
183  // The TX was used
184  _used = true;
185  // If the buffer is empty and available, write the byte directly
186  // and clear TXC to avoid malfunction on the UART
187  if ((_tx_buff_head == _tx_buff_tail) && _ctrl->stat.field.udre) {
188  ATOMIC({
189  *_data = byte;
190  _ctrl->stat.field.txc = true;
191  });
192  return 1;
193  }
194  // If the buffer has data we have to append the new byte after the
195  // last header index.
196  idx = (_tx_buff_head + 1) & UART_BUFF_MASK;
197  // If the TX buffer is full, wait for space
198  while (idx == _tx_buff_tail) {
199  // In the case that the interrupts are disabled, we have to
200  // poll the buffer ready flag and call the handler.
201  if (bit_is_clear(SREG, SREG_I)) {
202  if (_ctrl->stat.field.udre) _tx_callback();
203  }
204  }
205  // Append the new value to buffer
206  _tx_buff[idx] = byte;
207  // Update the head index and enable the interrupts
208  ATOMIC({
209  _tx_buff_head = idx;
210  _ctrl->ctrl1.field.udrie = true;
211  });
212  return 1;
213 }

228 inline void UART::_rx_callback(void) {
229  // Check status
230  if (!_ctrl->stat.field.upe) {
231  // No parity error, get data and update index
232  uint8_t new_head = (_rx_buff_head + 1) & UART_BUFF_MASK;
233  uint8_t data = *_data;
234  // Only store if the space is available!
235  if (new_head != _rx_buff_tail) {
236  _rx_buff[new_head] = data;
237  _rx_buff_head = new_head;
238  }
239  } else {
240  // Parity error, discard byte
241  *_data;
242  }
243 }

245 inline void UART::_tx_callback(void) {
246  // Check head/tail
247  if (_tx_buff_head != _tx_buff_tail) {
248  // Data available to send! Send more!
249  _tx_buff_tail = (_tx_buff_tail + 1) & UART_BUFF_MASK;
250  *_data = _tx_buff[_tx_buff_tail];
251  } else {
252  // No more data available. Stop interruptions!
253  _ctrl->ctrl1.field.udrie = false;
254  }
