101 bool DRV2605::calibrate(drv2605_bkfc_t brake_factor, drv2605_lpg_t loop_gain, drv2605_acalt_t cal_time,
102  uint8_t drive_time, bool startup_boost, bool brake_stabilizer, bool bidir_input ,
103  bool open_loop, bool pwm_input) {
104  uint8_t time = 0;
105  drv2605_reg_t reg;
106  // Set defaults
107  set_defaults();
108  // Initialize values
109  initialize_values(brake_factor, loop_gain, drive_time, startup_boost,
110  brake_stabilizer, bidir_input, open_loop, pwm_input);
111  // Set mode
112  set_register(0x07, DRV2605_REG_MODE); // Auto-calibration mode
113  // Set auto-calibration memory
114  update_field(reg, DRV2605_REG_CTRL4, ctrl4, A_CAL_TM, cal_time);
115  // Run auto-calibration
116  update_field(reg, DRV2605_REG_GO, go, GO, 1);
117 
118  // Wait until it's ready... GO = 0
119  // Maximum time: 2500[ms]
120  do {
121  delay_millis(10);
122  get_register(&reg.raw, DRV2605_REG_GO);
123  time++;
124  } while (reg.go.GO && (time < 250));
125 
126  // Check for errors
127  get_register(&reg.raw, DRV2605_REG_STAT);
128  if (reg.stat.DIAG_RES)
129  return false;
130 
131  // Get compensation, backEMF and feedback values
132  get_register(&_motor.acal_comp, DRV2605_REG_ACAL_COMP);
133  get_register(&_motor.acal_BEMF, DRV2605_REG_ACAL_BEMF);
134  get_register(&_motor.acal_feedback, DRV2605_REG_FB_CTRL);
135 
136  // Process complete
137  return true;
138 }

150 void DRV2605::initialize_values(drv2605_bkfc_t brake_factor, drv2605_lpg_t loop_gain, uint8_t drive_time,
151  bool startup_boost, bool brake_stabilizer, bool bidir_input,
152  bool open_loop, bool pwm_input) {
153  drv2605_reg_t reg;
154  // Set rated voltage
155  set_register(_motor.v_rat, DRV2605_REG_RATEDV);
156  // Set over-drive voltage
157  set_register(_motor.odc_lamp, DRV2605_REG_CLAMPV);
158  // Set feedback control
159  get_register(&reg.raw, DRV2605_REG_FB_CTRL);
160  reg.fb_ctrl.LP_GN = loop_gain;
161  reg.fb_ctrl.IS_LRA = _motor.is_LRA;
162  reg.fb_ctrl.BK_FCTR = brake_factor;
163  set_register(reg.raw, DRV2605_REG_FB_CTRL);
164  // Set control registers
165  // Control 1
166  get_register(&reg.raw, DRV2605_REG_CTRL1);
167  reg.ctrl1.DRV_TM = drive_time;
168  reg.ctrl1.ST_BST = startup_boost;
169  set_register(reg.raw, DRV2605_REG_CTRL1);
170  // Control 2
171  get_register(&reg.raw, DRV2605_REG_CTRL2);
172  reg.ctrl2.BDIR_IN = bidir_input;
173  reg.ctrl2.BK_STB = brake_stabilizer;
174  set_register(reg.raw, DRV2605_REG_CTRL2);
175  // Control 3
176  get_register(&reg.raw, DRV2605_REG_CTRL3);
177  reg.ctrl3.PWM_IN = !pwm_input;
178  if (_motor.is_LRA)
179  reg.ctrl3.LRA_O_LP = open_loop;
180  else
181  reg.ctrl3.ERM_O_LP = open_loop;
182  set_register(reg.raw, DRV2605_REG_CTRL3);
183 }
