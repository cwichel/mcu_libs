74 void TIMER8::init(uint32_t us, timer8_om_t mode) {
75  // Do this only if the mode selected implies custom periods.

77  case T8_OM_CTC_OCRA: case T8_OM_FPWM_OCRA: case T8_OM_CPWM_P_OCRA: {
78  _is_fixed = false;

120 void TIMER8::set_period(uint32_t us) {
121  // Only do this if the timer has custom periods.
122  if (_is_fixed)
123  return;
124  // Get the period length
125  const uint32_t cycles = ((_mode == T8_OM_CPWM_P_OCRA)? CYCLES_PER_US()/2 : CYCLES_PER_US()) * us;
126  // Set the best configuration
127  if (cycles < TIMER8_RESOLUTION) {
128  _clk_sel = T8_CS_1;
129  _period = cycles;
130  } else if (cycles < (TIMER8_RESOLUTION << 3)) {
131  _clk_sel = T8_CS_8;
132  _period = cycles >> 3;
133  } else if (cycles < (TIMER8_RESOLUTION << 6)) {
134  _clk_sel = T8_CS_64;
135  _period = cycles >> 6;
136  } else if (cycles < (TIMER8_RESOLUTION << 8)) {
137  _clk_sel = T8_CS_256;
138  _period = cycles >> 8;
139  } else if (cycles < (TIMER8_RESOLUTION << 10)) {
140  _clk_sel = T8_CS_1024;
141  _period = cycles >> 10;
142  } else {
143  _clk_sel = T8_CS_1024;
144  _period = TIMER8_RESOLUTION - 1;
145  }
146  // Set
147  _ctrl->ctrlb.field.cs = _clk_sel;
148  _cfg->ocra = _period;
149 }

235  // Define duty value
