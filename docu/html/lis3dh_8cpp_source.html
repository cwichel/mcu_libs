412 uint8_t LIS3DH::get_accel_conversion(void) {
413  uint8_t scale;
414  switch(_config.resolution) {
415  case LIS3DH_LOWPOWER: scale = 4; break; // 8bit mode: raw * 16;
416  case LIS3DH_NORMAL: scale = 2; break; // 10bit mode: raw * 4;
417  default: scale = 0; break; // 12bit mode: raw * 1;
418  }
419  switch(_config.scale) {
420  case LIS3DH_SCALE_4G: scale += 1; break; // 4G scale: 2G * 2;
421  case LIS3DH_SCALE_8G: scale += 2; break; // 8G scale: 2G * 4;
422  case LIS3DH_SCALE_16G: scale += 3; break; // 16G scale: 2G * 8;
423  default: scale += 0; break; // 2G scale;
424  }
425  return scale;
426 }
