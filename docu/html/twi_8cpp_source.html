73 uint16_t TWI::write(uint8_t addr, uint8_t* src, size_t size, bool stop) {
74  // Check if the buffer is available
75  if ((size == 0) || (src == NULL))
76  return 0x0005;
77  // Result aux.
78  uint16_t result;
79  // Wait until TWI is ready
80  while (_state != TWI_READY);
81  // Set TWI state machine
82  _state = TWI_MTX;
83  _error = TWI_ERR_NONE;
84  _send_stop = stop;
85  // Initialize master buffer vars.
86  _mx_buff = src;
87  _mx_buff_idx = 0;
88  _mx_buff_size = size;
89  // Build slave address
90  _slarw = (addr << 1) | TW_WRITE;
91  // Start TWI state machine
92  start();
93  // Wait until ready
94  while (_state == TWI_MTX);
95  // Get bytes transmitted
96  result = (_mx_buff_idx << 8);
97  // Handle errors
98  switch (_error) {
99  case TWI_ERR_NONE: result |= 0x00; break;
100  case TW_MT_SLA_NACK: result |= 0x01; break;
101  case TW_MT_DATA_NACK: result |= 0x02; break;
102  case TW_MT_ARB_LOST: result |= 0x03; break;
103  default: result |= 0x04; break;
104  }
105  // Return result
106  return result;
107 }

353 void TWI::state_machine(void) {
354  /* TWI state machine */
355  switch (TW_STATUS) {
356  /****************************************
357  * MASTER:
358  * 1.- Start / Rep. Start:
359  * 1.1.- Copy slave address to buffer.
360  * 1.2.- Send address (expect ACK).
361  ***************************************/
362  case TW_START:
363  case TW_REP_START: {
364  TWDR = _slarw;
365  reply(true);
366  } break;
367  /****************************************
368  * MASTER TRANSMITTER:
369  * 1.- Slave address/data sent, ACK received:
370  * 1.1.- If there's data to send (idx < len):
371  * 1.1.1.- Copy byte to buffer.
372  * 1.1.2.- Send byte (expect ACK).
373  * 1.2.- Otherwise, go to "Master End Transaction Handler"
374  ***************************************/
375  case TW_MT_SLA_ACK:
376  case TW_MT_DATA_ACK: {
377  if (_mx_buff_idx < _mx_buff_size) {
378  TWDR = _mx_buff[_mx_buff_idx++];
379  reply(true);
380  } else {
381  goto _tw_meth;
382  }
383  } break;
384  /****************************************
385  * MASTER RECEIVER:
386  * 1.- Slave address sent, ACK received:
387  * 1.1.- If data is expected (idx < len), send ACK (prepare bus for reception).
388  * 1.2.- Otherwise, send NACK.
389  ***************************************/
390  case TW_MR_SLA_ACK: {
391  _tw_mrgdh: // Master Receiver Get Data Handler
392  if (_mx_buff_idx < _mx_buff_size) {
393  reply(true);
394  } else {
395  reply(false);
396  }
397  } break;
398  /* 2.- Slave data received, ACK sent:
399  * 2.1.- Copy byte from buffer.
400  * 2.2.- Go to "Master Receiver Get Data Handler" (MR 1.-)
401  ***************************************/
402  case TW_MR_DATA_ACK: {
403  _mx_buff[_mx_buff_idx++] = TWDR;
404  goto _tw_mrgdh;
405  } break;
406  /* 3.- Slave data received, NACK sent:
407  * 3.1.- Copy final byte from buffer.
408  * 3.2.- Go to "Master End Transaction Handler"
409  ***************************************/
410  case TW_MR_DATA_NACK: {
411  _mx_buff[_mx_buff_idx] = TWDR;
412  goto _tw_meth;
413  } break;
414  /****************************************
415  * If "Bus Error" or "Slave address sent, NACK received" (MT/MR)
416  * or "Data sent, NACK received" (MT):
417  * 1.- Save error.
418  * 2.- Stop transaction.
419  ***************************************/
420  case TW_MT_SLA_NACK:
421  case TW_MR_SLA_NACK:
422  case TW_MT_DATA_NACK: {
423  _error = TW_STATUS;
424  stop();
425  } break;
426  /****************************************
427  * If "Arbitration lost" (MT/MR):
428  * 1.- Save error.
429  * 2.- Release bus.
430  ***************************************/
431  case TW_MT_ARB_LOST: {
432  _error = TW_STATUS;
433  release_bus();
434  } break;
435  /****************************************
436  * SLAVE TRANSMITTER
437  * 1.- Adressed / Arbitration lost, ACK sent:
438  * 1.1.- Enter slave TX mode (STX).
439  * 1.2.- Reset TX data size and index.
440  * 1.3.- Execute slave TX event: Must prepare data to send using
441  * twi_slave_write(uint8_t* data, size_t size). If data is
442  * not set, prepare to send 0x00.
443  * 1.4.- Go to "Slave Transmitter Send Data Handler" (ST 2.-)
444  ***************************************/
445  case TW_ST_SLA_ACK:
446  case TW_ST_ARB_LOST_SLA_ACK: {
447  _state = TWI::TWI_STX;
448  _stx_buff_idx = 0;
449  _stx_buff_size = 0;
450  on_request_callback();
451  if (_stx_buff_size == 0) {
452  _stx_buff_size = 1;
453  *_stx_buff = 0x00;
454  }
455  goto _tw_stsdh;
456  } break;
457  /* 2.- Data sent, ACK received:
458  * 2.1.- Put byte on buffer.
459  * 2.3.- If there's data to send (idx < len), send byte (expect ACK).
460  * 2.4.- Otherwise, send byte (expect NACK).
461  ***************************************/
462  case TW_ST_DATA_ACK: {
463  _tw_stsdh: // Slave Transmitter Send Data Handler
464  TWDR = _stx_buff[_stx_buff_idx++];
465  if (_stx_buff_idx < _stx_buff_size) {
466  reply(true);
467  } else {
468  reply(false);
469  }
470  } break;
471  /* 3.- Data sent / Last data sent, NACK received:
472  * 3.1.- ACK future responses.
473  * 3.2.- Leave slave TX state.
474  ***************************************/
475  case TW_ST_DATA_NACK:
476  case TW_ST_LAST_DATA: {
477  reply(true);
478  _state = TWI::TWI_READY;
479  } break;
480  /****************************************
481  * SLAVE RECEIVER
482  * 1.- Adressed / Arbitration lost, ACK sent:
483  * 1.1.- Enter slave RX mode (SRX).
484  * 1.2.- Reset RX data index.
485  * 1.3.- Prepare line for reception (send ACK).
486  ***************************************/
487  case TW_SR_SLA_ACK:
488  case TW_SR_GCALL_ACK:
489  case TW_SR_ARB_LOST_SLA_ACK:
490  case TW_SR_ARB_LOST_GCALL_ACK: {
491  _state = TWI::TWI_SRX;
492  _srx_buff_idx = 0;
493  reply(true);
494  } break;
495  /* 2.- Data received, ACK sent:
496  * 2.1.- If there's space on buffer:
497  * 2.1.1.- Copy byte from buffer.
498  * 2.1.2.- Send ACK (expect more data).
499  * 2.2.- Otherwise, send NACK.
500  ***************************************/
501  case TW_SR_DATA_ACK:
502  case TW_SR_GCALL_DATA_ACK: {
503  if (_srx_buff_idx < _srx_buff_size) {
504  _srx_buff[_srx_buff_idx++] = TWDR;
505  reply(true);
506  } else {
507  reply(false);
508  }
509  } break;
510  /* 3.- Data received, NACK sent:
511  * 3.1.- Send NACK.
512  ***************************************/
513  case TW_SR_DATA_NACK:
514  case TW_SR_GCALL_DATA_NACK: {
515  reply(false);
516  } break;
517  /* 4.- Stop / Rep. start received:
518  * 4.1.- Release bus.
519  * 4.2.- If there's space, put a termination value on Rx buffer ('\0').
520  * 4.3.- Execute slave RX event: Must handle input data using
521  * twi_slave_rx_event(uint8_t last_idx). Where last_idx points the
522  * last index used on the RX buffer.
523  * 4.4.- Reset RX buffer index.
524  ***************************************/
525  case TW_SR_STOP: {
526  release_bus();
527  if (_srx_buff_idx < _srx_buff_size) {
528  _srx_buff[_srx_buff_idx] = '\0';
529  }
530  on_receive_callback(_srx_buff_idx);
531  _srx_buff_idx = 0;
532  } break;
533  /****************************************
534  * If "Slave address sent, NACK received" (MT/MR)
535  * or "Data sent, NACK received" (MT):
536  * 1.- Save error.
537  * 2.- Stop transaction.
538  ***************************************/
539  case TW_BUS_ERROR: {
540  _error = TW_STATUS;
541  stop();
542  } break;
543  /****************************************
544  * If "No state information" (any):
545  * 1.- Do nothing.
546  ***************************************/
547  case TW_NO_INFO: {
548  } break;
549  }
550  /* End of ISR */
551  return;
552 
553  /********************************************
554  * Master End Transaction Handler
555  * 1.1.- If send_stop flag is active:
556  * 1.1.1.- Stop transaction.
557  * 1.2.- Otherwise, prepare rep. start.
558  * 1.2.1.- Put rep_start flag active.
559  * 1.2.2.- Send start but disable interruptions*(1).
560  * 1.2.3.- Set TWI state machine as ready.
561  ********************************************/
562  _tw_meth: {
563  if (_send_stop) {
564  stop();
565  } else {
566  _rep_start = true;
567  TWCR = _BV(TWEN) | _BV(TWINT) | _BV(TWSTA);
568  _state = TWI::TWI_READY;
569  }
570  } return;
571 }
