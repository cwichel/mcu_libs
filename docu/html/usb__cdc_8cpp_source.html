35 bool CDC_setup(USBSetup& setup) {
36  uint8_t r = setup.bRequest;
37  uint8_t requestType = setup.bmRequestType;
38 
39  if (REQUEST_DEVICETOHOST_CLASS_INTERFACE == requestType) {
40  if (CDC_GET_LINE_CODING == r) {
41  USB_send_control(0,(void*)&_usbLineInfo, 7);
42  return true;
43  }
44  }
45 
46  if (REQUEST_HOSTTODEVICE_CLASS_INTERFACE == requestType) {
47  if (CDC_SEND_BREAK == r) {
48  breakValue = ((uint16_t)setup.wValueH << 8) | setup.wValueL;
49  }
50  if (CDC_SET_LINE_CODING == r) {
51  USB_recv_control((void*)&_usbLineInfo, 7);
52  }
53  if (CDC_SET_CONTROL_LINE_STATE == r) {
54  _usbLineInfo.lineState = setup.wValueL;
55 
56  // auto-reset into the bootloader is triggered when the port, already
57  // open at 1200 bps, is closed. this is the signal to start the watchdog
58  // with a relatively long period so it can finish housekeeping tasks
59  // like servicing endpoints before the sketch ends
60  uint16_t magic_key_pos = MAGIC_KEY_POS;
61 
62  // If we don't use the new RAMEND directly, check manually if we have a newer bootloader.
63  // This is used to keep compatible with the old leonardo bootloaders.
64  // You are still able to set the magic key position manually to RAMEND-1 to save a few bytes for this check.
65  #if MAGIC_KEY_POS != (RAMEND-1)
66  // For future boards save the key in the inproblematic RAMEND
67  // Which is reserved for the main() return value (which will never return)
68  if (is_lufa_boot()) {
69  // horray, we got a new bootloader!
70  magic_key_pos = (RAMEND-1);
71  }
72  #endif
73 
74  // We check DTR state to determine if host port is open (bit 0 of lineState).
75  if (1200 == _usbLineInfo.dwDTERate && (_usbLineInfo.lineState & 0x01) == 0) {
76  #if MAGIC_KEY_POS != (RAMEND-1)
77  // Backup ram value if its not a newer bootloader and it hasn't already been saved.
78  // This should avoid memory corruption at least a bit, not fully
79  if (magic_key_pos != (RAMEND-1) && *(uint16_t *)magic_key_pos != MAGIC_KEY) {
80  *(uint16_t *)(RAMEND-1) = *(uint16_t *)magic_key_pos;
81  }
82  #endif
83  // Store boot key
84  *(uint16_t *)magic_key_pos = MAGIC_KEY;
85  // Save the watchdog state in case the reset is aborted.
86  wdtcsr_save = WDTCSR;
87  wdt_enable(WDTO_120MS);
88  } else if (*(uint16_t *)magic_key_pos == MAGIC_KEY) {
89  // Most OSs do some intermediate steps when configuring ports and DTR can
90  // twiggle more than once before stabilizing.
91  // To avoid spurious resets we set the watchdog to 120ms and eventually
92  // cancel if DTR goes back high.
93  // Cancellation is only done if an auto-reset was started, which is
94  // indicated by the magic key having been set.
95  wdt_reset();
96  // Restore the watchdog state in case the sketch was using it.
97  WDTCSR |= (1<<WDCE) | (1<<WDE);
98  WDTCSR = wdtcsr_save;
99 
100  #if MAGIC_KEY_POS != (RAMEND-1)
101  // Restore backed up (old bootloader) magic key data
102  if (magic_key_pos != (RAMEND-1)) {
103  *(uint16_t *)magic_key_pos = *(uint16_t *)(RAMEND-1);
104  } else
105  #endif
106  {
107  // Clean up RAMEND key
108  *(uint16_t *)magic_key_pos = 0x0000;
109  }
110  }
111  }
112  return true;
113  }
114  return false;
115 }

186 size_t USBSerial::write(uint8_t* buff, size_t size) {
187  /* only try to send bytes if the high-level CDC connection itself
188  is open (not just the pipe) - the OS should set lineState when the port
189  is opened and clear lineState when the port is closed.
190  bytes sent before the user opens the connection or after
191  the connection is closed are lost - just like with a UART. */
192 
193  // TODO - ZE - check behavior on different OSes and test what happens if an
194  // open connection isn't broken cleanly (cable is yanked out, host dies
195  // or locks up, or host virtual serial port hangs)
196  if (_usbLineInfo.lineState > 0) {
197  int16_t r = USB_send(CDC_TX, buff, size);
198  if (r > 0)
199  return r;
200  }
201  return 0;
202
