64 void adc_configure(adc_ref_t ref, bool hi_speed) {
65  // Set ADC reference (by default ADLAR is disabled and ADC0 is selected)
66  _admux->raw = 0x00;
67  _admux->field.ref = ref;
68  // Enable ADC and set the prescaler (by default ADC has no trigger and
69  // is not interrupt driven).
70  _ctrla->raw = 0x00;
71  _ctrla->field.adps = adc_prescaler();
72  // Set if the ADC will work on high speed (only if the mcu supports it.
73  // by default the ADC is in free running mode and the analog comparator
74  // multiplexer is disabled).
75  _ctrlb->raw = 0x00;
76  _ctrlb->field.adhsm = hi_speed;
77 }
