231  // If the target buffer or the word to find are empty, do nothing
232  if ((_len == 0) || (find._len == 0)) return;
233  // Otherwise, find and replace
234  int16_t str_diff = replace._len - find._len;
235  char* source = _buff;
236  char* found_at;
237  if (str_diff == 0) {
238  // If the old and new values have the same size...
239  while ((found_at = strstr(source, find._buff)) != NULL) {
240  memcpy(found_at, replace._buff, replace._len);
241  source = found_at + replace._len;
242  }
243  } else if (str_diff < 0) {
244  // If the new value is smaller than the old one...
245  char* output = _buff;
246  while ((found_at = strstr(source, find._buff)) != NULL) {
247  uint16_t pos = found_at - source;
248  memcpy(output, source, pos);
249  output += pos;
250  memcpy(output, replace._buff, replace._len);
251  output += replace._len;
252  source += pos + find._len;
253  _len += str_diff;
254  }
255  strcpy(output, source);
256  } else {
257  // If the new value is bigger than the old one...
258  uint16_t new_len = _len;
259  while ((found_at = strstr(source, find._buff)) != NULL) {
260  source = found_at + find._len;
261  new_len += str_diff;
262  }
263  if (new_len == _len) return;
264  if ((new_len > _cap) && !buffer_change(new_len)) return;
265  int16_t index = _len - 1;
266  while ((index > 0) && ((index = last_index_of(find, index)) >= 0)) {
267  source = _buff + index + find._len;
268  memmove(source + str_diff, source, _len - (source - _buff));
269  memcpy(_buff + index, replace._buff, replace._len);
