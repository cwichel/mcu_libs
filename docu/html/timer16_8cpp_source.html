86 void TIMER16::set_period(uint32_t us) {
87  // Get the period length
88  const uint32_t cycles = ((_mode == T16_OM_CPWM_PF)? CYCLES_PER_US()/2 : CYCLES_PER_US()) * us;
89  // Set the best configuration
90  if (cycles < TIMER16_RESOLUTION) {
91  _clk_sel = T16_CS_1;
92  _period = cycles;
93  } else if (cycles < (TIMER16_RESOLUTION << 3)) {
94  _clk_sel = T16_CS_8;
95  _period = cycles >> 3;
96  } else if (cycles < (TIMER16_RESOLUTION << 6)) {
97  _clk_sel = T16_CS_64;
98  _period = cycles >> 6;
99  } else if (cycles < (TIMER16_RESOLUTION << 8)) {
100  _clk_sel = T16_CS_256;
101  _period = cycles >> 8;
102  } else if (cycles < (TIMER16_RESOLUTION << 10)) {
103  _clk_sel = T16_CS_1024;
104  _period = cycles >> 10;
105  } else {
106  _clk_sel = T16_CS_1024;
107  _period = TIMER16_RESOLUTION - 1;
108  }
109  // Set
110  _ctrl->ctrlb.field.cs = _clk_sel;
111  _cfg->icr = _period;
112 }

188  // Define duty value
